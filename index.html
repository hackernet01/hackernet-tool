async function verifyUID() { const uid = document.getElementById('uidInput').value.trim(); if (!uid || uid.length !== 6) { showMessage('Enter valid 6-digit UID', 'error'); return; } try { const response = await fetch(`/api/check.js?uid=${uid}`); const data = await response.json(); if (data.valid) { currentUID = uid; grantAccess(uid, data); // Send Telegram notification const ip = await getUserIP(); const message = `🔓 UID ACCESS\nUID: ${uid}\nName: ${data.name}\nIP: ${ip}\nTime: ${new Date().toLocaleString()}`; sendToTelegram(message); } else { showMessage('❌ Invalid UID or Expired. Contact admin.', 'error'); } } catch (error) { showMessage('Error verifying UID. Try again.', 'error'); } }
