function approveUID() { const uid = document.getElementById('approveUID').value.trim(); const name = document.getElementById('approveName').value.trim(); const days = parseInt(document.getElementById('approveDays').value); if (!uid || uid.length !== 6) { showPopup('❌ Enter valid 6-digit UID', 'error'); return; } fetch('/api/approve.js', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ uid, name, days, adminKey: 'hackernet123' }) }) .then(res => res.json()) .then(data => { if (data.success) { showPopup(`✅ UID ${uid} approved for ${days} days!`, 'success'); document.getElementById('approveUID').value = ''; document.getElementById('approveName').value = ''; loadAllData(); } else { showPopup(`❌ Approval failed: ${data.error}`, 'error'); } }) .catch(err => { showPopup('❌ Network error', 'error'); }); }
